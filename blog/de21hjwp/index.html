链表排序

约 1298 字大约 4 分钟

2026-04-22

链表排序算法可分为 比较排序 和 非比较排序 两大类，每种算法都有其适用场景和性能特点。

小规模数据：选择冒泡排序或插入排序

大规模数据：优先考虑归并排序

特定场景：根据数据特征选择相应的非比较排序

比较排序算法

算法核心思想时间复杂度空间复杂度特点
冒泡排序相邻节点比较交换O(n2)O(1)实现简单，适合小规模数据
选择排序选择最小节点交换O(n2)O(1)交换次数少，适合交换成本高的场景
插入排序逐个插入到有序序列O(n2)O(1)链表上表现优异，适合部分有序数据
归并排序分治合并O(nlog⁡n)O(1)链表上的最优选择，稳定高效
快速排序基准分割递归O(nlog⁡n)O(1)平均性能好，但最坏情况 O(n2)O(n2)

非比较排序算法

算法核心思想时间复杂度空间复杂度适用条件
计数排序统计频率重建O(n+k)O(k)值域范围小，k 为值域大小
桶排序分桶排序合并O(n)O(n+m)数据分布均匀，m 为桶数
基数排序按位分桶排序O(n×d)O(n+k)数字位数 d 较小

算法适用性分析

适用性排序算法说明
完全适合冒泡排序、选择排序、插入排序、归并排序、快速排序天然适合链表结构
需要额外空间计数排序、桶排序、基数排序需要辅助数组，但算法逻辑适合
不适合希尔排序（需要访问序列中第 i+gap个元素，链表无法直接跳转，必须从头遍历，时间复杂度从 O(1) 退化为 O(n)依赖随机访问，链表无法高效实现
不推荐堆排序(基于完全二叉树结构，数组可以通过下标 O(1)访问父子节点，而链表需要遍历查找，效率极低。)完全二叉树结构用链表存储效率低

冒泡排序

时间复杂度 o(n2)

class Solution {
    public ListNode sortList(ListNode head) {
	if (head == null || head.next == null) {
			return head;
		}
		//定义循环初始节点
		ListNode strNode = head;
		//定义循环结束节点
		ListNode endNode = null;
		while (strNode.next != null) {
			ListNode nodeA = head;
			while (nodeA.next != null && nodeA != endNode) {
                //如果当前节点的值大于后一节点的值，就让他们交换值
				if (nodeA.val >nodeA.next.val) {
					int maxVal = nodeA.val;
					nodeA.val = nodeA.next.val;
					nodeA.next.val = maxVal;
				}
				nodeA = nodeA.next;
			}
            //内层循环结束，让当前节点等于循环结束节点，
			endNode = nodeA;
			strNode = strNode.next;
		}
		return head;
    }
}
选择排序

时间复杂度 o(n2)

class Solution {
    public ListNode sortList(ListNode head) {
        ListNode strNode = head;
        
        while (strNode != null && strNode.next != null) {

            ListNode minNode = strNode;

            ListNode nextNode = strNode.next;
			//内循环是为了找出带有最小值的节点
            while (nextNode != null) {
                if (minNode.val > nextNode.val) {
                    minNode = nextNode;
                }
                nextNode = nextNode.next;
            }
            //将当前节点和最小值节点的值交换
            if (strNode != minNode) {
                int minVal = minNode.val;
                minNode.val = strNode.val;
                strNode.val = minVal;
            }
            strNode = strNode.next;
        }
        return head;
    }
}

归并排序

class Solution {
    public ListNode sortList(ListNode head) {
        if (head == null || head.next == null) {
			return head;
		}
		ListNode slow = head;
		ListNode fast = head.next;
        //这里是为了快速找到中间位置
		while (fast != null && fast.next != null) {
			slow = slow.next;
			fast = fast.next.next;
		}
		ListNode left = head;
		ListNode right = slow.next;
		slow.next = null;
		Solution s = new Solution();
        //递归调用
		return s.merge(sortList(left), sortList(right));
    }
    
	public ListNode merge(ListNode left,ListNode right){
		ListNode head = new ListNode(-1);
		ListNode cur = head;
		//left和right都不为空的情况下
		while(left!=null&&right!=null){
            //left 小于 right，将left节点接到cur后面, left 变为 right ，第二次循环的时候就可以把 right接到cur后面了
			if (left.val < right.val) {
				cur.next = left;
				left = left.next;
			}else {
				cur.next = right;
				right = right.next;
			}
			cur= cur.next;
		}
        
		if (left!=null) {
			cur.next = left;
		}
		if (right!=null) {
			cur.next = right;
		}
		return head.next;

	}
 	
}

	//快速排序
	public ListNode quickSort(ListNode head,ListNode next){
		ListNode left = head;
		ListNode right = next;
		//递归终止条件：区间长度小于等于 1
		if (left == right||left.next == right) {
			return left;
		}
		bubbleSort bs = new bubbleSort();
		//分割链表，获取基准值位置
		ListNode partition = bs.partition(left, right);
		bs.quickSort(left,partition);
		bs.quickSort(right,partition.next);

		return left;

	}

	public ListNode partition(ListNode left,ListNode right){

		if (left == right|| left.next == right){
			return left;
		}
		//使用快慢指针进行分割
		int pivot = left.val;
		//node_i: 指向小于基准值的最后一个节点
		ListNode node_i = left;
		//node_j: 遍历指针，寻找小于基准值的节点
		ListNode node_j = left.next;
		while (node_j != right ) {
			//发现一个小于基准值的元素
			if (node_j.val < pivot) {
				//将 node_i 向右移动一位
				node_i = node_i.next;
				//交换 node_i 和 node_j 的值，保证 node_i 之前的节点都小于基准值
				int tmp = node_i.val;
				node_i.val = node_j.val;
				node_j.val = tmp;
			}
			node_j = node_j.next;
		}
		//将基准节点放到正确位置上（node_i 位置）
		int tmp = node_i.val;
		node_i.val = left.val;
		left.val = tmp;
		return node_i;
	}
